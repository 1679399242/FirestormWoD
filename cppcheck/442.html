            }

        void Cleanup(Creature* infernal, InfernalPoint *point)
        {
            if (infernals.empty())
                return;

            for (std::vector<uint64>::iterator itr = infernals.begin(); itr!= infernals.end(); ++itr)
                if (*itr == infernal->GetGUID())
            {
                infernals.erase(itr);
                break;
            }

            positions.push_back(point);
        }
    };

};
